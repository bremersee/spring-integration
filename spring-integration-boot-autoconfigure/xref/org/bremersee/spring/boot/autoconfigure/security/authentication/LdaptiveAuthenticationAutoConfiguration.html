60  /**
61   * The ldaptive authentication auto-configuration.
62   *
63   * @author Christian Bremer
64   */
65  @AutoConfiguration
66  @ConditionalOnClass(name = {
67      "org.ldaptive.ConnectionFactory",
68      "org.bremersee.ldaptive.LdaptiveTemplate",
69      "org.bremersee.spring.security.ldaptive.authentication.LdaptiveAuthenticationManager"
70  })
71  @ConditionalOnBean({ConnectionConfig.class})
72  @ConditionalOnProperty(prefix = "bremersee.authentication.ldaptive", name = "user-base-dn")
73  @AutoConfigureAfter({LdaptiveAutoConfiguration.class})
74  @EnableConfigurationProperties(AuthenticationProperties.class)
75  @Slf4j
76  public class LdaptiveAuthenticationAutoConfiguration {
77  
78    private final LdaptiveAuthenticationProperties properties;
79  
80    /**
81     * Instantiates a new ldaptive authentication autoconfiguration.
82     *
83     * @param properties the properties
84     */
85    public LdaptiveAuthenticationAutoConfiguration(AuthenticationProperties properties) {
86      this.properties = LdaptivePropertiesMapper.map(properties);
87    }
88  
89    /**
90     * Init.
91     */
92    @EventListener(ApplicationReadyEvent.class)
93    public void init() {
94      log.info("""
95              
96              *********************************************************************************
97              * {}
98              * properties = {}
99              *********************************************************************************""",
100         ClassUtils.getUserClass(getClass()).getSimpleName(),
101         properties);
102   }

104   /**
105    * Creates ldaptive password encoder provider.
106    *
107    * @return the ldaptive password encoder provider
108    */
109   @ConditionalOnMissingBean
110   @Bean
111   public LdaptivePasswordEncoderProvider ldaptivePasswordEncoderProvider() {
112     return LdaptivePasswordEncoderProvider.defaultProvider();
113   }
114 
115   /**
116    * The ldaptive remember-me token provider.
117    *
118    * @param accountControlEvaluator the account control evaluator
119    * @return the ldaptive remember-me token provider
120    */
121   @ConditionalOnMissingBean
122   @Bean
123   public LdaptiveRememberMeTokenProvider ldaptiveRememberMeTokenProvider(
124       ObjectProvider<AccountControlEvaluator> accountControlEvaluator) {
125 
126     AccountControlEvaluator evaluator = accountControlEvaluator
127         .getIfAvailable(() -> properties.getAccountControlEvaluator().get());
128     if (!isEmpty(properties.getPasswordLastSetAttribute())) {
129       return new LdaptivePwdLastSetRememberMeTokenProvider(
130           evaluator, properties.getPasswordLastSetAttribute());
131     }
132     return new LdaptiveEvaluatedRememberMeTokenProvider(evaluator);
133   }
134 
135   /**
136    * Creates ldaptive authentication manager.
137    *
138    * @param connectionConfig the connection config
139    * @param connectionFactoryProvider the connection factory provider
140    * @param ldaptiveTemplateProvider the ldaptive template provider
141    * @param ldaptivePasswordEncoderProvider the ldaptive password encoder provider
142    * @param ldaptiveRememberMeTokenProvider the ldaptive remember-me token provider
143    * @param emailToUsernameResolver the email to username resolver
144    * @param usernameToBindDnConverter the username to bind dn provider
145    * @param accountControlEvaluator the account control evaluator
146    * @param grantedAuthoritiesMapper the granted authorities mapper
147    * @param tokenConverter the token converter
148    * @return the ldaptive authentication manager
149    */
150   @ConditionalOnWebApplication(type = Type.SERVLET)
151   @Bean(initMethod = "init")
152   public LdaptiveAuthenticationManager ldaptiveAuthenticationManager(
153       ConnectionConfig connectionConfig,
154       ObjectProvider<ConnectionFactory> connectionFactoryProvider,
155       ObjectProvider<LdaptiveTemplate> ldaptiveTemplateProvider,
156       LdaptivePasswordEncoderProvider ldaptivePasswordEncoderProvider,
157       LdaptiveRememberMeTokenProvider ldaptiveRememberMeTokenProvider,
158       ObjectProvider<EmailToUsernameResolver> emailToUsernameResolver,
159       ObjectProvider<UsernameToBindDnConverter> usernameToBindDnConverter,
160       ObjectProvider<AccountControlEvaluator> accountControlEvaluator,
161       ObjectProvider<GrantedAuthoritiesMapper> grantedAuthoritiesMapper,
162       ObjectProvider<Converter<LdaptiveUserDetails, LdaptiveAuthentication>> tokenConverter) {
163 
164     LdaptiveAuthenticationManager manager = new LdaptiveAuthenticationManager(
165         getLdaptiveTemplate(connectionConfig, connectionFactoryProvider, ldaptiveTemplateProvider),
166         properties);
167     manager.setPasswordEncoder(ldaptivePasswordEncoderProvider.get());
168     manager.setPasswordProvider(ldaptiveRememberMeTokenProvider);
169     emailToUsernameResolver.ifAvailable(manager::setEmailToUsernameResolver);
170     usernameToBindDnConverter.ifAvailable(manager::setUsernameToBindDnConverter);
171     accountControlEvaluator.ifAvailable(manager::setAccountControlEvaluator);
172     manager.setGrantedAuthoritiesMapper(getGrantedAuthoritiesMapper(grantedAuthoritiesMapper));
173     tokenConverter.ifAvailable(manager::setTokenConverter);
174     return manager;
175   }
176 
177   /**
178    * Creates reactive ldaptive authentication manager.
179    *
180    * @param connectionConfig the connection config
181    * @param connectionFactoryProvider the connection factory provider
182    * @param ldaptiveTemplateProvider the ldaptive template provider
183    * @param ldaptivePasswordEncoderProvider the ldaptive password encoder provider
184    * @param ldaptiveRememberMeTokenProvider the ldaptive remember-me token provider
185    * @param emailToUsernameResolver the email to username resolver
186    * @param usernameToBindDnConverter the username to bind dn converter
187    * @param accountControlEvaluator the account control evaluator
188    * @param grantedAuthoritiesMapper the granted authorities mapper
189    * @param tokenConverter the token converter
190    * @return the reactive ldaptive authentication manager
191    */
192   @ConditionalOnWebApplication(type = Type.REACTIVE)
193   @Bean
194   public ReactiveLdaptiveAuthenticationManager reactiveLdaptiveAuthenticationManager(
195       ConnectionConfig connectionConfig,
196       ObjectProvider<ConnectionFactory> connectionFactoryProvider,
197       ObjectProvider<LdaptiveTemplate> ldaptiveTemplateProvider,
198       LdaptivePasswordEncoderProvider ldaptivePasswordEncoderProvider,
199       LdaptiveRememberMeTokenProvider ldaptiveRememberMeTokenProvider,
200       ObjectProvider<EmailToUsernameResolver> emailToUsernameResolver,
201       ObjectProvider<UsernameToBindDnConverter> usernameToBindDnConverter,
202       ObjectProvider<AccountControlEvaluator> accountControlEvaluator,
203       ObjectProvider<GrantedAuthoritiesMapper> grantedAuthoritiesMapper,
204       ObjectProvider<Converter<LdaptiveUserDetails, LdaptiveAuthentication>> tokenConverter) {
205     return new ReactiveLdaptiveAuthenticationManager(
206         ldaptiveAuthenticationManager(
207             connectionConfig,
208             connectionFactoryProvider,
209             ldaptiveTemplateProvider,
210             ldaptivePasswordEncoderProvider,
211             ldaptiveRememberMeTokenProvider,
212             emailToUsernameResolver,
213             usernameToBindDnConverter,
214             accountControlEvaluator,
215             grantedAuthoritiesMapper,
216             tokenConverter));
217   }
218 
219   private LdaptiveTemplate getLdaptiveTemplate(
220       ConnectionConfig connectionConfig,
221       ObjectProvider<ConnectionFactory> connectionFactoryProvider,
222       ObjectProvider<LdaptiveTemplate> ldaptiveTemplateProvider) {
223 
224     LdaptiveTemplate ldaptiveTemplate = ldaptiveTemplateProvider.getIfAvailable();
225     if (nonNull(ldaptiveTemplate)) {
226       return ldaptiveTemplate;
227     }
228     ConnectionFactory connectionFactory = connectionFactoryProvider.getIfAvailable();
229     if (nonNull(connectionFactory)) {
230       return new LdaptiveTemplate(connectionFactory);
231     }
232     return new LdaptiveTemplate(new DefaultConnectionFactory(connectionConfig));
233   }
234 
235   private GrantedAuthoritiesMapper getGrantedAuthoritiesMapper(
236       ObjectProvider<GrantedAuthoritiesMapper> grantedAuthoritiesMapper) {
237     return grantedAuthoritiesMapper.getIfAvailable(() -> new NormalizedGrantedAuthoritiesMapper(
238         properties.getDefaultRoles(),
239         properties.toRoleMappings(),
240         properties.getRolePrefix(),
241         properties.getRoleCaseTransformation(),
242         properties.toRoleStringReplacements()));
243   }
