package org.bremersee.minio;

/**
 * The minio object id interface.
 *
 * @author Christian Bremer
 */
public interface MinioObjectId {

  /**
   * From object name.
   *
   * @param name the name
   * @return the minio object id
   */
  static MinioObjectId from(String name) {
    return from(name, null);
  }

  /**
   * From object name and version ID.
   *
   * @param name the name
   * @param version the version
   * @return the minio object id
   */
  static MinioObjectId from(String name, String version) {
    return new DefaultMinioObjectId(name, version);
  }

  /**
   * Gets name.
   *
   * @return the name
   */
  String getName();

  /**
   * Gets version id.
   *
   * @return the version id
   */
  String getVersionId();

}
