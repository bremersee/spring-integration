19  class LdaptiveRememberMeAutoConfigurationTest {
20  
21    private final AuthenticationProperties properties = new AuthenticationProperties();
22  
23    private LdaptiveRememberMeAutoConfiguration target;
24  
25    /**
26     * Init.
27     */
28    @BeforeEach
29    void init() {
30      properties.getRememberMe().setKey("secret");
31      properties.getLdaptive().setUserBaseDn("ou=users,dc=system");
32  
33      properties.getLdaptive().setTemplate(Template.OPEN_LDAP);
34      target = new LdaptiveRememberMeAutoConfiguration(properties);
35      target.init();
36  
37      properties.getLdaptive().setTemplate(Template.ACTIVE_DIRECTORY);
38      target = new LdaptiveRememberMeAutoConfiguration(properties);
39      target.init();
40    }
41  
42    /**
43     * Remember me authentication provider.
44     */
45    @Test
46    void rememberMeAuthenticationProvider() {
47      RememberMeAuthenticationProvider actual = target.rememberMeAuthenticationProvider();
48      assertThat(actual).isNotNull();
49    }
50  
51    /**
52     * Remember me services.
53     */
54    @Test
55    void rememberMeServices() {
56      LdaptiveAuthenticationManager authenticationManager = mock(LdaptiveAuthenticationManager.class);
57      LdaptiveUserDetailsService userDetailsService = mock(LdaptiveUserDetailsService.class);
58      doReturn(userDetailsService).when(authenticationManager).getUserDetailsService();
59      RememberMeServices services = target.rememberMeServices(authenticationManager);
60      assertThat(services).isNotNull();
61    }
62  
63    /**
64     * Remember me authentication filter.
65     */
66    @Test
67    void rememberMeAuthenticationFilter() {
68      LdaptiveAuthenticationManager authenticationManager = mock(LdaptiveAuthenticationManager.class);
69      LdaptiveUserDetailsService userDetailsService = mock(LdaptiveUserDetailsService.class);
70      doReturn(userDetailsService).when(authenticationManager).getUserDetailsService();
71      RememberMeAuthenticationFilter actual = target.rememberMeAuthenticationFilter(
72          authenticationManager, target.rememberMeServices(authenticationManager));
73      assertThat(actual).isNotNull();
74    }
