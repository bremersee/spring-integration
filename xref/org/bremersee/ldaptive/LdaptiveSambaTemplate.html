35  /**
36   * The ldaptive samba template.
37   *
38   * @author Christian Bremer
39   */
40  public class LdaptiveSambaTemplate extends LdaptiveTemplate {
41  
42    private Supplier<String> passwordGenerator;
43  
44    /**
45     * Instantiates a new ldaptive samba template.
46     *
47     * @param connectionFactory the connection factory
48     */
49    public LdaptiveSambaTemplate(ConnectionFactory connectionFactory) {
50      super(connectionFactory);
51      this.passwordGenerator = getDefaultPasswordGenerator();
52    }
53  
54    /**
55     * Sets password generator.
56     *
57     * @param passwordGenerator the password generator
58     */
59    public void setPasswordGenerator(Supplier<String> passwordGenerator) {
60      if (Objects.nonNull(passwordGenerator)) {
61        this.passwordGenerator = passwordGenerator;
62      }
63    }
64  
65    @Override
66    public LdaptiveSambaTemplate clone(LdaptiveErrorHandler errorHandler) {
67      LdaptiveSambaTemplate template = new LdaptiveSambaTemplate(getConnectionFactory());
68      template.setErrorHandler(errorHandler);
69      return template;
70    }
71  
72    @Override
73    public String generateUserPassword(String dn) {
74      String newPassword = passwordGenerator.get();
75      setUserPassword(dn, newPassword);
76      return newPassword;
77    }
78  
79    private void setUserPassword(String dn, String newPassword) {
80      String quotedPassword = "\"" + newPassword + "\"";
81      char[] unicodePwd = quotedPassword.toCharArray();
82      byte[] pwdArray = new byte[unicodePwd.length * 2];
83      for (int i = 0; i < unicodePwd.length; i++) {
84        pwdArray[i * 2 + 1] = (byte) (unicodePwd[i] >>> 8);
85        pwdArray[i * 2] = (byte) (unicodePwd[i] & 0xff);
86      }
87      LdapAttribute ldapAttribute = new LdapAttribute();
88      ldapAttribute.setName("unicodePwd");
89      ldapAttribute.setBinary(true);
90      ldapAttribute.addBinaryValues(pwdArray);
91      AttributeModification attributeModification = new AttributeModification(
92          Type.REPLACE, ldapAttribute);
93      ModifyRequest modifyRequest = ModifyRequest.builder()
94          .dn(dn)
95          .modifications(attributeModification)
96          .build();
97      this.clone(new AbstractLdaptiveErrorHandler() {
98            @Override
99            public LdaptiveException map(LdapException ldapException) {
100             int httpStatus;
101             String errorCode;
102             String reason;
103             if (ldapException.getResultCode() == ResultCode.CONSTRAINT_VIOLATION
104                 && ldapException.getMessage().contains("check_password_restrictions")) {
105               httpStatus = 400;
106               errorCode = "check_password_restrictions";
107               reason = "New password restrictions were violated.";
108             } else if (ldapException.getResultCode() == ResultCode.NO_SUCH_OBJECT) {
109               httpStatus = 404;
110               reason = "No such object found.";
111               errorCode = "org.bremersee.ldaptive:26cfda51-0fde-443c-87b1-502fc165bb94";
112             } else {
113               httpStatus = 500;
114               reason = "Internal error.";
115               errorCode = "org.bremersee.ldaptive:a70939fb-2c94-412f-80c0-00a7d5dcf4a6";
116             }
117             return LdaptiveException.builder()
118                 .httpStatus(httpStatus)
119                 .errorCode(errorCode)
120                 .reason(reason)
121                 .cause(ldapException)
122                 .build();
123           }
124         })
125         .modify(modifyRequest);
126   }
127 
128   @Override
129   public void modifyUserPassword(String dn, String oldPass, String newPass) {
130     if (bind(new SimpleBindRequest(dn, oldPass))) {
131       setUserPassword(dn, newPass);
132     } else {
133       throw LdaptiveException.builder()
134           .reason("Bind failed.")
135           .errorCode("bind_failed")
136           .httpStatus(400)
137           .build();
138     }
139   }
140 
141   private static Supplier<String> getDefaultPasswordGenerator() {
142     PasswordGenerator pwGen = new PasswordGenerator();
143     CharacterData special = new CharacterData() {
144       @Override
145       public String getErrorCode() {
146         return "INSUFFICIENT_SPECIAL";
147       }
148 
149       @Override
150       public String getCharacters() {
151         return "!#$%&'()*+,-./:;<=>?@[]_{|}~";
152       }
153     };
154     List<CharacterRule> rules = List.of(
155         // at least five upper-case character
156         new CharacterRule(EnglishCharacterData.UpperCase, 5),
157 
158         // at least five lower-case character
159         new CharacterRule(EnglishCharacterData.LowerCase, 5),
160 
161         // at least three digit character
162         new CharacterRule(EnglishCharacterData.Digit, 3),
163 
164         // at least one special character
165         new CharacterRule(special, 1));
166     return () -> pwGen.generatePassword(16, rules);
167   }
