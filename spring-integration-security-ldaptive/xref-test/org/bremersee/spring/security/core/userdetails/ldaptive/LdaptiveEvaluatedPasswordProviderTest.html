13   * The ldaptive evaluated remember-me token provider test.

15  class LdaptiveEvaluatedPasswordProviderTest {
16  
17    private LdaptiveEvaluatedRememberMeTokenProvider target;
18  
19    /**
20     * Init.
21     */
22    @BeforeEach
23    void init() {
24      target = new LdaptiveEvaluatedRememberMeTokenProvider();
25      target.setAccountControlEvaluator(new NoAccountControlEvaluator());
26    }
27  
28    /**
29     * Gets remember-me token.
30     */
31    @Test
32    void getRememberMeToken() {
33      String dn = "cn=test,ou=users,dc=bremersee,dc=org";
34      LdapEntry ldapEntry = mock(LdapEntry.class);
35      doReturn(dn).when(ldapEntry).getDn();
36      String evaluation = "true:true:true:true-";
37      assertThat(target.getRememberMeToken(ldapEntry))
38          .isEqualTo(evaluation + dn);
39    }
