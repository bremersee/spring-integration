26  /**
27   * The ldaptive user details service test.
28   */
29  @ExtendWith(SoftAssertionsExtension.class)
30  class LdaptiveUserDetailsServiceTest {
31  
32    private static final String USER_BASE_DN = "ou=people,dc=bremersee,dc=org";
33    private static final String USER_DN = "uid=junit,ou=people,dc=bremersee,dc=org";
34  
35    private LdaptiveTemplate ldaptiveTemplate;
36  
37    /**
38     * Init ldaptive user details service.
39     *
40     * @param properties the properties
41     * @return the ldaptive user details service
42     */
43    LdaptiveUserDetailsService init(LdaptiveAuthenticationProperties properties) {
44      ldaptiveTemplate = mock(LdaptiveTemplate.class);
45      LdaptiveUserDetailsService target = new LdaptiveUserDetailsService(
46          properties, ldaptiveTemplate);
47      target.setRememberMeTokenProvider(null); // has no effect
48      target.setAccountControlEvaluator(null); // has no effect
49      target.setGrantedAuthoritiesMapper(null); // has no effect
50      return target;
51    }
52  
53    /**
54     * Load not existing user.
55     */
56    @Test
57    void loadNotExistingUser() {
58      LdaptiveAuthenticationProperties properties = new UserContainsGroupsTemplate();
59      properties.setUserBaseDn(USER_BASE_DN);
60      properties.setUserRdnAttribute("uid");
61      LdaptiveUserDetailsService target = init(properties);
62  
63      doReturn(Optional.empty()).when(ldaptiveTemplate).findOne(any());
64  
65      assertThatExceptionOfType(UsernameNotFoundException.class)
66          .isThrownBy((() -> target.loadUserByUsername("junit")));
67    }
68  
69    /**
70     * Load user with groups by username.
71     *
72     * @param softly the softly
73     */
74    @Test
75    void loadUserWithGroupsByUsername(SoftAssertions softly) {
76      LdaptiveAuthenticationProperties properties = new UserContainsGroupsTemplate();
77      properties.setUserBaseDn(USER_BASE_DN);
78      properties.setUserRdnAttribute("uid");
79      LdaptiveUserDetailsService target = init(properties);
80  
81      LdapEntry user = createUser();
82      doReturn(Optional.of(user)).when(ldaptiveTemplate).findOne(any());
83  
84      LdaptiveUserDetails actual = target.loadUserByUsername(USER_DN);
85  
86      assertThat(actual).isNotNull();
87      assertActual(actual, softly);
88    }
89  
90    /**
91     * Load user and groups by username.
92     *
93     * @param softly the softly
94     */
95    @Test
96    void loadUserAndGroupsByUsername(SoftAssertions softly) {
97      LdaptiveAuthenticationProperties properties = new GroupContainsUsersTemplate();
98      properties.setUserBaseDn(USER_BASE_DN);
99      properties.setGroupBaseDn(USER_BASE_DN);
100     properties.setUserRdnAttribute("uid");
101     properties.setGroupIdAttribute("uid"); // doesn't exist -> use rdn
102     properties.setGroupMemberFormat("${username}");
103 
104     LdaptiveUserDetailsService target = init(properties);
105 
106     LdapEntry user = createUser();
107     doReturn(Optional.of(user)).when(ldaptiveTemplate).findOne(any());
108 
109     LdapEntry group = createGroup();
110     doReturn(List.of(group)).when(ldaptiveTemplate).findAll(any());
111 
112     LdaptiveUserDetails actual = target.loadUserByUsername("junit");
113 
114     assertThat(actual).isNotNull();
115     assertActual(actual, softly);
116   }
117 
118   private void assertActual(LdaptiveUserDetails actual, SoftAssertions softly) {
119     softly
120         .assertThat(actual.getDn())
121         .isEqualTo(USER_DN);
122     softly
123         .assertThat(actual.getName())
124         .isEqualTo("junit");
125     softly
126         .assertThat(actual.getUsername())
127         .isEqualTo("junit");
128     softly
129         .assertThat(actual.getPassword())
130         .isNotEmpty();
131     List<GrantedAuthority> actualAuthorities = new ArrayList<>(actual.getAuthorities());
132     List<GrantedAuthority> expectedAuthorities = List.of(new SimpleGrantedAuthority("ROLE_tester"));
133     softly
134         .assertThat(actualAuthorities)
135         .containsExactlyInAnyOrderElementsOf(expectedAuthorities);
136     softly
137         .assertThat(actual.isEnabled())
138         .isTrue();
139     softly
140         .assertThat(actual.isAccountNonLocked())
141         .isTrue();
142     softly
143         .assertThat(actual.isAccountNonExpired())
144         .isTrue();
145     softly
146         .assertThat(actual.isCredentialsNonExpired())
147         .isTrue();
148   }
149 
150   private LdapEntry createUser() {
151     LdapEntry entry = new LdapEntry();
152     entry.setDn(USER_DN);
153     entry.addAttributes(LdapAttribute.builder().name("uid").values("junit").build());
154     entry.addAttributes(LdapAttribute.builder().name("givenName").values("Test").build());
155     entry.addAttributes(LdapAttribute.builder().name("sn").values("User").build());
156     entry.addAttributes(LdapAttribute.builder().name("mail").values("junit@example.com").build());
157     entry.addAttributes(LdapAttribute.builder()
158         .name("memberOf").values("cn=tester," + USER_BASE_DN).build());
159     return entry;
160   }
161 
162   private LdapEntry createGroup() {
163     LdapEntry entry = new LdapEntry();
164     entry.setDn("cn=tester," + USER_BASE_DN);
165     entry.addAttributes(LdapAttribute.builder()
166         .name("uniqueMember").values("junit").build());
167     return entry;
168   }
