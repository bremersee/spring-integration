3   import java.util.ArrayList;
4   import java.util.Collection;
5   import java.util.List;
6   import java.util.Map;
7   import java.util.Set;
8   import org.assertj.core.api.Assertions;
9   import org.junit.jupiter.params.ParameterizedTest;
10  import org.junit.jupiter.params.provider.ValueSource;
11  import org.springframework.security.core.GrantedAuthority;
12  import org.springframework.security.core.authority.SimpleGrantedAuthority;
13  
14  /**
15   * The normalized granted authorities mapper test.
16   */
17  class NormalizedGrantedAuthoritiesMapperTest {
18  
19    /**
20     * Map authorities.
21     *
22     * @param transformation the transformation
23     */
24    @ParameterizedTest
25    @ValueSource(strings = {"TO_UPPER_CASE", "TO_LOWER_CASE"})
26    void mapAuthorities(String transformation) {
27      CaseTransformation caseTransformation = CaseTransformation.valueOf(transformation);
28      NormalizedGrantedAuthoritiesMapper target = createTarget(caseTransformation);
29      Collection<? extends GrantedAuthority> authorities = List.of(
30          new SimpleGrantedAuthority("junit-developers"),
31          new SimpleGrantedAuthority("junit-developers"),
32          new SimpleGrantedAuthority("foo"));
33      Collection<GrantedAuthority> actual = new ArrayList<>(target.mapAuthorities(authorities));
34      String normalizedValue = CaseTransformation.TO_UPPER_CASE.equals(caseTransformation)
35          ? "JUNIT_DEVELOPERS"
36          : "junit_developers";
37      Assertions
38          .assertThat(actual)
39          .containsExactlyInAnyOrderElementsOf(Set.of(
40              new SimpleGrantedAuthority("ROLE_" + normalizedValue),
41              new SimpleGrantedAuthority("ROLE_BAR"),
42              new SimpleGrantedAuthority("ROLE_LDAP")));
43    }
44  
45    private NormalizedGrantedAuthoritiesMapper createTarget(CaseTransformation caseTransformation) {
46      return new NormalizedGrantedAuthoritiesMapper(
47          List.of("ROLE_LDAP"),
48          Map.of("foo", "ROLE_BAR"),
49          "ROLE_",
50          caseTransformation,
51          Map.of("[-]", "_"));
52    }
53  }
